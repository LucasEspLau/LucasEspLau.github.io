Descripción general

El presente proyecto denominado "Safe Case" se centra en la creación de un prototipo de caja fuerte, el cual trabajaría a través de sensores y Wifi, los cuales nos permitiran cerrar y asegurar la caja fuerte a distancia, ademas detectar si alguna persona tuvo contacto con la caja fuerte y en el LCD se mostrara la contraseña colocada. El prototipo ayudaría a los clientes a saber si alguien a querido acceder a la caja fuerte y mantener seguro los objetos valiosos dentro de la caja.

Materiales

Para poder llevar a cabo nuestro proyecto en torno al uso de sensores utilizamos los siguientes materiales clasificados en:

1.-Componentes Electrónicos:

  • ESP8266
  • Placa de pruebas pequeña
  • 1 sensores de distancia ultrasonico
  • Sensor Infrarrojo
  • Servomotor
  • LCD 16x2
  • Interfaz I2C para el LCD
  • Cables H-M

2.-Materiales no Eléctricos

  • Por decidir

            #include

              #include 
              #define UBIDOTS_TOKEN "BBFF-oIP6xDzOC8M43Ly9hLPADRFwujHrGH"
              #define DEVICE_LABEL "jc-main-esp8266"
              #define LAB_DISTANCIA "distancia"
              #define LAB_SER "servo"
              #define LAB_INFRA "infra"
              #define LAB_CONTRA "contra"
              #include 
              LiquidCrystal_I2C lcd(0x27, 16, 2);
              
              const char* WIFI_SSID = "LUCAS";
              const char* WIFI_PASS = "73035832";
              const String contra="1234.00";
              int trigger1=D5, echo1=D6, servo=D0, infrarrojo=D7;
              long tiempo1;
              double  distancia1;
              int serv,estadoC, valorInfra;
              String contraOnline, persona, estado;
              Ubidots ubidots(UBIDOTS_TOKEN, UBI_HTTP);
              Servo puerta;
              
              void setup() {
                Serial.begin(115200);
                ubidots.wifiConnect(WIFI_SSID, WIFI_PASS);
                lcd.init();
                lcd.backlight();
                lcd.clear();
                pinMode(trigger1, OUTPUT);
                pinMode(echo1, INPUT);
                pinMode(infrarrojo, INPUT);
                serv=180;
                puerta.attach(servo);
                distancia1=0;
                valorInfra=0;
                contraOnline="caja";
                estado="cerrado";
                estadoC=0;
                puerta.write(serv);
              
              }
              
              void loop() {
                
                digitalWrite(trigger1,LOW); delayMicroseconds(2);
                digitalWrite(trigger1,HIGH); delayMicroseconds(10);
                digitalWrite(trigger1,LOW);   
              
                tiempo1= pulseIn(echo1,HIGH);
                distancia1= tiempo1/58.82;
                
                valorInfra=digitalRead(infrarrojo);
                
                ubidots.add(LAB_DISTANCIA, distancia1);
                ubidots.add(LAB_INFRA, valorInfra);
                
                bool bufferSent = false;
                bufferSent = ubidots.send(DEVICE_LABEL);
                if(bufferSent){
                  Serial.print("Valores enviados");
                }
                if(distancia1<=30){
                  persona="Accediendo";
                  estadoC=1;
                }else{
                  persona="Nadie accede";
                  estadoC=0;
                }
                
                Serial.println(distancia1);
                Serial.println(contraOnline);
                Serial.println(contra);
                Serial.println(estadoC);
                delay(500);
                
                  
                if(estadoC>0){
                  //En cercania
                  lcd.clear();
                  lcd.setCursor(0,0);
                  lcd.print("Safe Case");
                  lcd.setCursor(0,1);
                  lcd.print(persona);
                  delay(2000);
                
                  lcd.clear();
                  lcd.print("Hola, bienvenido a Safe Case");
                  puerta.write(serv);
                    for(int i=0;i<22; i++){
                        lcd.scrollDisplayLeft();
                        delay(250);
                    }
                    
                  lcd.clear();
                  lcd.setCursor(0,0);
                  lcd.print(persona);
                  lcd.setCursor(0,1);
                  lcd.print("Contraseña?");
                  contraOnline=String(ubidots.get(DEVICE_LABEL , LAB_CONTRA));
                    if(contraOnline.compareTo(contra)==0){
                      serv=ubidots.get(DEVICE_LABEL , LAB_SER);
                      puerta.write(serv);
                        if(serv<=180&serv>0&valorInfra==0){
                          estado="cerrada";
                        }else{
                          estado="abierta";
                        }
                      lcd.clear();
                      lcd.setCursor(0,0);
                      lcd.print("Correcto");
                      lcd.setCursor(0,1);
                      lcd.print("Puerta "+estado);  
                      Serial.println(serv);
                      Serial.println(contraOnline);
                    }else{
                      serv=180;
                      estado="cerrada";
                      lcd.clear();
                      lcd.setCursor(0,0);
                      lcd.print("Incorrecto");
                      lcd.setCursor(0,1);
                      lcd.print("Puerta "+estado);
                      Serial.println(serv);
                      Serial.println(contraOnline);
                    }
                }else{
                  //A distancia
                  lcd.clear();
                  lcd.setCursor(0,0);
                  lcd.print("Safe Case");
                  lcd.setCursor(0,1);
                  lcd.print(persona);
                  delay(2000);
                  contraOnline=String(ubidots.get(DEVICE_LABEL , LAB_CONTRA));
                  if(contraOnline.compareTo(contra)==0){
                    serv=ubidots.get(DEVICE_LABEL , LAB_SER);
                    puerta.write(serv);
                      if(serv<=180&serv>0&valorInfra==0){
                          estado="cerrada";
                      }else{
                          estado="abierta";
                      }
                    lcd.clear();
                    lcd.setCursor(0,0);
                    lcd.print("Correcto");
                    lcd.setCursor(0,1);
                    lcd.print("Puerta "+estado);  
                    Serial.println(serv);
                    Serial.println(contraOnline);
                  }else{
                    serv=180;
                    estado="cerrada";
                    lcd.clear();
                    lcd.setCursor(0,0);
                    lcd.print("Incorrecto");
                    lcd.setCursor(0,1);
                    lcd.print("Puerta "+estado);
                    Serial.println(serv);
                    Serial.println(contraOnline);
                  }
                  
                }
              }
